  /** âœ… Fetch chat details (returns existing chat or creates a new one) */
  async getChatDetails(userId: string, partnerId: string): Promise<Chat | null> {
    try {
      if (!userId || !partnerId) {
        console.error("âŒ Error: Invalid user ID or partner ID.");
        return null;

      // ğŸ” Check if a chat already exists between the users
      const { data: chatData, error: fetchError } = await supabase
        .from('chats')
        .select('id, created_at, updated_at')
        .or(`(user1_id.eq.${userId}.and.user2_id.eq.${partnerId}), (user1_id.eq.${partnerId}.and.user2_id.eq.${userId})`)
        .single();

      if (fetchError && fetchError.code !== 'PGRST116') {
        console.error('âŒ Error fetching chat:', fetchError);
        return null;
      }

      // âœ… If chat exists, return it
      if (chatData) {
        console.log("âœ… Existing chat found:", chatData.id);
        return { id: chatData.id, createdAt: chatData.created_at, updatedAt: chatData.updated_at, messages: [] };
      }

      // ğŸ†• No chat found â†’ Create a new one
      console.log("ğŸ†• No chat found, creating a new one...");
      const newChatId = await chatService.createChat(userId, partnerId);

      return { id: newChatId, createdAt: new Date().toISOString(), updatedAt: new Date().toISOString(), messages: [] };
    } catch (error) {
      console.error('âŒ Error in getChatDetails:', error);
      return null;
    }
  },

  /** âœ… Create a new chat if none exists */
  async createChat(userId: string, partnerId: string): Promise<string | null> {
    try {
      const { data: newChat, error } = await supabase
        .from('chats')
        .insert([
          {
            user1_id: userId,
            user2_id: partnerId,
            created_at: new Date().toISOString(),
            updated_at: new Date().toISOString(),
          }
        ])
        .select('id')
        .single();

    }
